int a[N] = {3, 4, 5, 1, 2};
int get_min(int *a, int len) {
    if(a == NULL || len <= 0) {
        return -1;
    }
    int p1 = 0, p2 = len-1;
    int mid = p1;
    
    while(a[p1] >= a[p2]) {
        if(p2 - p1 == 1) {
            mid = p2; break;
        }
        mid = (p1 + p2) / 2;
        if(a[p1] <= a[mid]) {
            p1 = mid;
        }
        else if(a[p2] >= a[mid]) {
            p2 = mid;
        }
    }
    return mid;
}

Node* swap(Node* head, Node* p, Node* q) {
    if(head == NULL || p == NULL || q == NULL) { return NULL; }
    if(p == q) return head;
    if(p->next == q) {
        Node* pre_p = findPre(head, p);
        pre_p->next = q;
        p->next = q->next;
        q->next = p;
    }
    else if(q->next == p) {
        Node* pre_q = findPre(head, q);
        pre_q->next = p;
        q->next = p->next;
        p->next = q;
    }
    else if(p != q) {
        Node* pre_p = findPre(head, p);
        Node* pre_q = findPre(head, q);
        Node* after_p = p->next;
        q->next = after_p;
        pre_p->next = q;
        pre_q->next = p;
    }
    return head;
}

int fdep(Node* root) {
    deque<Node*> dq;
    int high = -1;
    while(1) {
        for(; root != NULL; root = root->lchild) {
            dq.push_back(root);
        }
        high = max(high, (int)dq.size());
        while(1) {
            if(dq.empty()) return high;
            Node* par = dq.back();
            Node* rchild = par->rchild;
            if(rchild && root != rchild) {
                root = rchild;
                break;
            }
            root = par;
            dq.pop_back();
        }
    }
    return high;
}

    for(int i = 0; i < n; i++) {

    if(n == num) {
        cnt++;
        return;
    }

```cpp
void getMaxMin(int *a, int low, int high, int& maxv, int& minv) {  
    int mid, max1, max2, min1, min2;  
    if(high - low == 1 || high - low == 0) {  
        a[low] > a[high] ? (maxv = a[low], minv = a[high]) : (maxv = a[high], minv = a[low]);  
    }  
    else {  
        mid = (high + low) / 2;  
        getMaxMin(a, low, mid, max1, min1);  
        getMaxMin(a, mid+1, high, max2, min2);  
        maxv = max2 > max1 ? max2 : max1;  
        minv = min2 > min1 ? min1 : min2;  
    }  
}
```
